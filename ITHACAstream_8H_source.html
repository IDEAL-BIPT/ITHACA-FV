276 void ITHACAstream::SaveSparseMatrix(Eigen::SparseMatrix<T, whatever, IND>& m, word folder, word MatrixName)
277 {
278  typedef Eigen::Triplet<int> Trip;
279  std::vector<Trip> res;
280  m.makeCompressed();
281  mkDir(folder);
282 
283  std::fstream writeFile;
284  writeFile.open(folder + MatrixName, std::ios::binary | std::ios::out);
285 
286  if (writeFile.is_open())
287  {
288  IND rows, cols, nnzs, outS, innS;
289  rows = m.rows() ;
290  cols = m.cols() ;
291  nnzs = m.nonZeros() ;
292  outS = m.outerSize();
293  innS = m.innerSize();
294  writeFile.write(reinterpret_cast<const char *> (&rows), sizeof(IND));
295  writeFile.write(reinterpret_cast<const char *> (&cols), sizeof(IND));
296  writeFile.write(reinterpret_cast<const char *> (&nnzs), sizeof(IND));
297  writeFile.write(reinterpret_cast<const char *> (&outS), sizeof(IND));
298  writeFile.write(reinterpret_cast<const char *> (&innS), sizeof(IND));
299  writeFile.write(reinterpret_cast<const char *>(m.valuePtr()), sizeof(T ) * m.nonZeros());
300  writeFile.write(reinterpret_cast<const char *>(m.outerIndexPtr()), sizeof(IND) * m.outerSize());
301  writeFile.write(reinterpret_cast<const char *>(m.innerIndexPtr()), sizeof(IND) * m.nonZeros());
302  writeFile.close();
303  }

306 template <typename T, int whatever, typename IND>
307 void ITHACAstream::ReadSparseMatrix(Eigen::SparseMatrix<T, whatever, IND>& m, word folder, word MatrixName)
308 {
309  std::fstream readFile;
310  readFile.open(folder + MatrixName, std::ios::binary | std::ios::in);
311  if (!readFile.good())
312  {
313  std::cout << folder + MatrixName << " file does not exist, try to rerun the Offline Stage!" << std::endl;
314  exit(EXIT_FAILURE);
315  }
316  else if (readFile.is_open())
317  {
318  IND rows, cols, nnz, inSz, outSz;
319  readFile.read(reinterpret_cast<char *>(&rows ), sizeof(IND));
320  readFile.read(reinterpret_cast<char *>(&cols ), sizeof(IND));
321  readFile.read(reinterpret_cast<char *>(&nnz ), sizeof(IND));
322  readFile.read(reinterpret_cast<char *>(&inSz ), sizeof(IND));
323  readFile.read(reinterpret_cast<char *>(&outSz), sizeof(IND));
324  m.resize(rows, cols);
325  m.makeCompressed();
326  m.resizeNonZeros(nnz);
327  readFile.read(reinterpret_cast<char *>(m.valuePtr()) , sizeof(T ) * nnz );
328  readFile.read(reinterpret_cast<char *>(m.outerIndexPtr()), sizeof(IND) * outSz);
329  readFile.read(reinterpret_cast<char *>(m.innerIndexPtr()), sizeof(IND) * nnz );
330  m.finalize();
331  readFile.close();
332  }

337 void ITHACAstream::SaveDenseMatrix(MatrixType& Matrix, word folder, word MatrixName)
338 {
339  mkDir(folder);
340  std::ofstream out(folder + MatrixName, std::ios::out | std::ios::binary | std::ios::trunc);
341  typename MatrixType::Index rows = Matrix.rows(), cols = Matrix.cols();
342  out.write(reinterpret_cast<char *> (&rows), sizeof(typename MatrixType::Index));
343  out.write(reinterpret_cast<char *> (&cols), sizeof(typename MatrixType::Index));
344  out.write(reinterpret_cast<char *> (Matrix.data()), rows * cols * sizeof(typename MatrixType::Scalar) );
345  out.close();
346 }
